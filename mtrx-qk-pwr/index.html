矩阵快速幂

引入|快速幂

我们知道，求 an(a,n∈N+)a^{n} (a,n \in N+)an(a,n∈N+) 时，当 nnn 的第 iii 位（从低到高，i≥0i \geq 0i≥0）为1时，ana^{n}an的拆解表达式里就要乘上一项a2ia^{2^{i}}a2i

C语言描述如下：

int a, n, ans;
ans = 1;
while (n > 0)
{
    if (n % 2) ans *= a;
    a *= a;
    n /= 2;
}

该算法的时间复杂度取决于 nnn 的二进制形式有多少位，因此 T(n)=⌈log2(n)⌉=O(logn)T(n) = \lceil log_2(n) \rceil = O(logn)T(n)=⌈log2​(n)⌉=O(logn)

推导|矩阵快速幂

快速幂算法解决的是整数求幂的问题，而矩阵快速幂解决的是矩阵求幂的问题，两者没有本质区别。矩阵快速幂算法常用于解决线性递推式的加速。
对于矩阵快速幂，首先要知道单位矩阵。

[10⋯001⋯0⋮⋮⋱⋮00⋯1] \begin{bmatrix} 1 & 0 & \cdots & 0 \\ 0 & 1 & \cdots & 0 \\ \vdots & \vdots & \ddots & \vdots \\ 0 & 0 & \cdots & 1 \end{bmatrix} ⎣⎢⎢⎢⎡​10⋮0​01⋮0​⋯⋯⋱⋯​00⋮1​⎦⎥⎥⎥⎤​

明显，单位矩阵与任何矩阵相乘，都会得到该矩阵，这个矩阵就是对矩阵进行快速幂时候的基础矩阵。
矩阵快速幂是ICPC中对于求递推式能用到的模板，能实现 O(N3logM)O(N^{3} log M)O(N3logM) 的时间复杂度（其中 N是矩阵阶乘，M是要求的第几项）。

C语言描述如下：

typedef struct matrix
{
    int m[MAXN][MAXN];
}matrix;
matrix ans, res;
matrix mul (matrix, matrix, int);
void pow (int, int);

matrix mul (matrix a, matrix b, int n)
{
    matrix tmp;
    int i, j, k;
    for (i = 1; i <= n; ++i)
    	for (j = 1; j <= n; ++j) tmp.m [i][j] = 0;
    	
    for (i = 1; i <= n; ++i)
		for (j = 1; j <= n; ++j)
			for (k = 1; k <= n; ++k) tmp.m[i][j] += a.m[i][k] * b.m[k][j];
			
	return tmp;		
}
void pow (int N, int n)
{
	int i, j;
	for (i = 1; i <= n; ++i)
		for (j = 1; j <= n; ++j)
			if (i == j) ans.m[i][j] = 1;
			else ans.m[i][j] = 0;
	
	while (N)
	{
		if (N & 1) ans = mul (ans, res);
		res = mul (res, res);
		N >>= 1;	
	}	
}

C语言描述如下：

#define MAXN 2
#define mod 10000

typedef struct matrix
{
    int m[MAXN][MAXN];
}matrix;

void init (matrix);
matrix mul (matrix, matrix, int);
matrix pow (matrix, int);

void init (matrix a)
{
	int i;
	memset (a.m, 0, sizeof (a.m));
	for (i = 0; i < MAXN; ++i) a[i][i] = 1;
}
matrix mul (matrix a, matrix b)
{
    matrix tmp;
    int i, j, k, n = MAXN;
    for (i = 1; i <= n; ++i)
    	for (j = 1; j <= n; ++j) tmp.m [i][j] = 0;
    	
    for (i = 1; i <= n; ++i)
		for (j = 1; j <= n; ++j)
			for (k = 1; k <= n; ++k)
			{
				tmp.m[i][j] += a.m[i][k] * b.m[k][j];
				tmp.m[i][j] %= mod;
			}
			
	return tmp;		
}
matrix pow (matrix res, int N)
{
	matrix ans;
	init (ans);
	while (N)
	{
		if (N & 1) ans = mul (ans, res);
		res = mul (res, res);
		N >>= 1;	
	}	
	return ans;
}
