int power (int base, int pow) //å¿«é€Ÿå¹‚ 
{
    int result = 1;
    while (pow > 0) 
	{
        if (pow & 1) result = result * base;
        pow >>= 1;
        base = base * base;
    }
    return result;
}

	int denominator; //åˆ†æ¯
	int reduc_frac () //çº¦åˆ† 
	{
		int temp = gcd (numerator, denominator);

int reduc_com (frac a, frac b) //é€šåˆ† 
{
	int temp = lcm (a.denominator, b.denominator);
	a.numerator *= (temp / a.denominator);
	b.numerator *= (temp / a.denominator);
	a.denominator = b.denominator = temp;
	return 1;
}
frac operator + (frac a, frac b) //åˆ†æ•°åŠ æ³• 
{
	frac ans;
	reduc_com (a, b);
	ans.denominator = a.denominator;
	ans.numerator = a.numerator + b.numerator;
	ans.reduc_frac ();
	return ans;
}
frac operator - (frac a, frac b) //åˆ†æ•°å‡æ³• 
{
	frac ans;
	reduc_com (a, b);
	ans.denominator = a.denominator;
	ans.numerator = a.numerator - b.numerator;
	ans.reduc_frac ();
	return ans;
}
frac operator * (frac a, frac b) //åˆ†æ•°ä¹˜æ³• 
{
	frac ans;
	ans.denominator = a.denominator * b.denominator;
	ans.numerator = a.numerator * b.numerator;
	ans.reduc_frac ();
	return ans;
}
frac operator / (frac a, frac b) //åˆ†æ•°é™¤æ³• 
{
	cswap (b.denominator, b.numerator);
	return a * b;
}
frac operator ^ (frac n, int k)
{
	frac ans;
	ans.numerator = power (n.numerator, k);
	ans.denominator = power (n.denominator, k);
	return ans;

string plus (string a, string b) //å­—ç¬¦ä¸²åŠ æ³• 
{
    string ans;
    int len1 = a.length (); int len2 = b.length ();
    if (len1 < len2) for (int i = 1; i <= len2 - len1; ++i) a = "0" + a;
    else for (int i = 1; i <= len1 - len2; ++i) b = "0" + b;
    len1 = a.length ();
    int cf = 0, temp;
    for (int i = len1 - 1; i >= 0; --i)
    {
        temp = a[i] - '0' + b[i] - '0' + cf;
        cf = temp / 10; temp %= 10;
        ans = char (temp + '0') + ans;
    }
    if (cf != 0) ans = char (cf + '0') + ans;

string minus (string a, string b) //å­—ç¬¦ä¸²å‡æ³• 
{
    string str;
    int pr = 0;
    if (a < b) 
    {
        swap (a, b); pr = -1;
    }
    int tmp = a.length () - b.length(), cf = 0;
    for (int i = b.length () - 1; i >= 0; --i)
    {
        if (b[tmp + i] < b[i] + cf)
        {
            str = char (a[tmp + i] - b[i] - cf + '0' + 10) + str;
            cf = 1;
        }
        else
        {
            str = char (a[tmp + i] - b[i] - cf + '0') + str;
            cf = 0;
        }
    }
    for (int i = tmp - 1; i >= 0; --i)
    {
        if (a[i] - cf >= '0')
        {
            str = char (a[i] - cf) + str;
            cf = 0;
        }
        else
        {
            str = char (a[i] - cf + 10) + str;
            cf = 1;
        }
    }
    str.erase (0, str.find_first_not_of ('0'));
    if (str.size () == 0) 
    {
        str = "0";
        pr = -1;
    }
    if (pr == 0) str = "-" + str;
    return str;
}
string times (string a, string b) //å­—ç¬¦ä¸²ä¹˜æ³• 
{
    int len1 = a.length (), len2 = b.length ();
    string str, tempstr;
    for (int i = len2 - 1; i >= 0; --i)
    {
        tempstr = "";
        int temp = b[i] - '0', t = 0, cf = 0;
        if (temp != 0)
        {
            for (int j = 1; j <= len2 - 1 - i; ++j) tempstr += "0";
            for (int j = len1 - 1; j >= 0; --j)
            {
                t = (temp * (a[j] - '0') + cf) % 10;
                cf = (temp * (a[j] - '0') + cf) / 10;
                tempstr = char (t + '0') + tempstr;
            }
            if (cf != 0) tempstr = char (cf + '0') + tempstr;

    string ans;

        int n = 0;
        string A = a.substr (0, b.length ());

        {
            string t = minus (A, b);
            if (t >= "0")
